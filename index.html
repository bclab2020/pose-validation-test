// EVAL_MODULES 内の "alignment_front" を以下のように拡張・設計変更 alignment_front: (ctx, getK, canvasObj) => { ctx.fillStyle = "#deff9a"; ctx.fillText("🧍 静止姿勢: 前面（重心動揺解析）", 30, 45); const rS = getK('right_shoulder'), lS = getK('left_shoulder'); const rA = getK('right_ankle'), lA = getK('left_ankle'); const nose = getK('nose'); // 1. 既存の肩傾斜度の表示 if (rS && lS) { const tilt = ((lS.y - rS.y) / (lS.x - rS.x) * 180 / Math.PI); ctx.fillStyle = "#ffffff"; ctx.fillText(`肩傾斜度: ${tilt.toFixed(1)}°`, 30, 85); } // 2. 【新規】垂直正中線に対する「頭部の側方偏位・動揺」のリアルタイム可視化 if (rA && lA && nose) { const midAnkleX = (lA.x + rA.x) / 2; // 垂直線からのミリメートル換算（※カメラ距離に応じたピクセル比。テストベッドとして倍率4を乗算） const headSwayX = (nose.x - midAnkleX) * 4; ctx.fillStyle = headSwayX < 0 ? "#ff5252" : "#00ff88"; ctx.fillText(`頭部偏位: ${headSwayX > 0 ? "右" : "左"} ${Math.abs(headSwayX).toFixed(1)} mm`, 30, 125); // 3. 【視覚化】頭部位置にリアルタイムの動揺インジケーター（矢印とライン）を描画 ctx.strokeStyle = "rgba(255, 255, 0, 0.6)"; ctx.lineWidth = 1.5; ctx.setLineDash([4, 4]); ctx.beginPath(); ctx.moveTo(midAnkleX, nose.y); ctx.lineTo(nose.x, nose.y); ctx.stroke(); ctx.setLineDash([]); // 現在のブレを中心点としてマーク ctx.fillStyle = "yellow"; ctx.beginPath(); ctx.arc(nose.x, nose.y, 4, 0, 2*Math.PI); ctx.fill(); } }
